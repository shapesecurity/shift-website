Demo

The program on the left is rendered on the right. Mousing over any identifier will highlight it all other identifiers which refer to the same variable, color-coded by type of access (declaration, read, or write).

function bytes(n, maxBytes) { if (maxBytes == null) { maxBytes = Math.max(1, Math.ceil(Math.log2(n + 1) / 8)); } var rv = Array(maxBytes); for (var bIndex = maxBytes - 1; bIndex >= 0; --bIndex) { rv[bIndex] = n & 0xFF; n >>>= 8; } return rv; }
